- Install sass:
    + Run npm init (to create package )
    + npm install gulp gulp-sass sass --save-dev (to install gulp,evironment replace sass to css)
    + Create file "gulpfile.js"
    + This is the content in gulpfile.js


const { src, dest, watch , series} = require('gulp');
const sass = require('gulp-sass')(require('sass'));

function buildStyles(){

        return src('*.scss')
        .pipe(sass())
        .pipe(dest('css'))
}

function watchTask(){
    watch(['*.scss'],buildStyles)
}
exports.default = series(buildStyles,watchTask)
